  /* 引数の簡易チェック */
  if(argc != 3) {
    return EXIT_FAILURE;
  }

  int32_t shift = atoi(argv[2]) % 26;
  /* マイナスにも対応 */
  if(shift < 0) { shift += 26; }

    /* 大文字の場合 */
    if(isupper(argv[1][i])) {
      printf("%c", (argv[1][i] - 'A' + shift) % 26 + 'A');
    }
    /* 小文字の場合 */
    else if (islower(argv[1][i])) {
      printf("%c", (argv[1][i] - 'a' + shift) % 26 + 'a');
    }
    /* 記号とかの場合 */

      printf("%c", argv[1][i]);
    }
  }
  return EXIT_SUCCESS;
}

指定した値で文字列を分割し、それぞれ逆にして出力することがわかった。

ここで、問題の文字列を眺めてみると、最初に大文字の Y があり、その前に ruo があることから、Your になると推測できる。
